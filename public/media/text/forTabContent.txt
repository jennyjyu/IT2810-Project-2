this.state = {
        animalLimericks: [],
        test: ""

    }

getText() {
     fetch("./media/text/animals.json")
         .then(response => response.json())
         .then(response => {
            const {limericks} = response.data
            this.setState({ animalLimericks: limericks })
            this.setState({ test: limericks[0].text })
        })
 }


 renderText() {
     var moreText = []
     var newText = this.state.animalLimericks[0].text.split("\r")
     for (var i = 0; i<newText.length; i++) {
         moreText.push(<p>{newText[i]}</p>)
     }
     return moreText
 }
